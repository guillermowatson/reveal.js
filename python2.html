Creando clases

Tenemos un ejemplo en persona.py


#!/usr/bin/python
# coding: utf-8

class Persona:
    """Abstracción de los objetos Persona"""
    def __init__(self, nombre):
        self.nombre = nombre
        print("Acabo de ser creado y me llamo {nombre}").format(nombre=nombre)

    def avanzar(self):
        print("Soy {nombre} y estoy caminando hacia adelante!").format(nombre=self.nombre)
Hacer el ejemplo con el IDE: Eclipse.
__init__: Método especial --> se llama al instanciar la clase.
self: convención! --> es una referencia.

Encapsulando


class Encapsulamiento:
    def __init__(self):
        self.__atributo_privado = "este atributo es privado."
        self._atributo_semi_privado = "este atributo es \"casi\" privado."
        self.atributo_publico = "este atributo es público."

    def publico(self):
        return "Este es un método Público"

    def _semi_privado(self):
        return "Este es un método Semi Privado"

    def __privado(self):
        return "Este es un método Privado"

    def get_atributo_privado(self):
        return self.__atributo_privado

    def set_atributo_privado(self, valor):
        self.__atributo_privado = valor

    secreto = property(get_atributo_privado, set_atributo_privado)

Herencia

Tenemos un ejemplo en trabajador.py 


class Trabajador:
    def __init__(self, nombre):
        self.nombre = nombre
        
    def trabajar(self):
        print("-{name}: Trabajo muy duro! Como un esclavo!").format(name=self.nombre)


class Jubilado(Trabajador):
    def trabajar(self):
        print("-{name}: Ya no trabajo! Soy JUBILADO!!").format(name=self.nombre)


class Menor(Trabajador):
    def trabajar(self):
        print("-{name}: EXPLOTADOR!!!!").format(name=self.nombre)

Herencia Múltiple

Tenemos un ejemplo en multiple.py


import trabajador

class Persona:
    def __init__(self, edad):
        self.edad =  edad
        
    def trabajar(self):
        print("-Persona: tengo {age} años.").format(age=self.edad)
        

class Empleado(Persona, trabajador.Trabajador):
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad
        
    def identidad(self):
        print("Soy {name} y tengo {age} años").format(name=self.nombre, age=self.edad)

Métodos Especiales

  • __init__: tareas de inicialización.
  • __new__: constructor (clase).
  • __del__: destructor del objeto (limpieza).
  • __str__: lo que usa print.
  • __cmp__: comparador entre objetos.
  • __len__: longitud del objeto.

#!/usr/bin/python
# coding: utf-8

class MiError(Exception):
    def __init__(self, valor):
        self.valor = valor
        
    def __str__(self):
        return "Error: " + str(self.valor)

try:
    valor = input("Ingrese un número:")
    if valor > 0:
        raise MiError("Ingresó un número positivo!")
except MiError, e:
    print e

print("Programa finalizado")
